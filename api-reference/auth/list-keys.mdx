---
title: "List API Keys"
api: "GET /api/v1/auth/keys"
description: "Get all API keys for the current user"
---

Get all API keys for the current user.

Requires JWT Bearer authentication with the token received from the login endpoint.

### Headers

<ParamField header="Authorization" type="string" required>
  Bearer token (JWT) from OAuth2 login
</ParamField>

### Response

<ResponseField name="keys" type="array">
  Array of API key objects
  
  <Expandable title="API Key Object">
    <ResponseField name="id" type="integer">
      Unique identifier for the API key
    </ResponseField>
    
    <ResponseField name="key_prefix" type="string">
      First few characters of the API key (e.g., "sk-1234")
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Optional name for the API key
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp of creation
    </ResponseField>
    
    <ResponseField name="is_active" type="boolean">
      Whether the key is currently active
    </ResponseField>
    
    <ResponseField name="is_default" type="boolean">
      Whether this is the default key
    </ResponseField>
  </Expandable>
</ResponseField>
